137 private async UniTask ExplodingViewTask()
138 {
139 if (transform.position != lastPosition || transform.rotation != lastRotation)
140 {
141 UpdateExplodables();
142 lastPosition = transform.position;
143 lastRotation = transform.rotation;
144 }
145 UniTask[] tasks = new UniTask[Explodables.Count];
146 for (int i = 0; i < Explodables.Count; i++)
147 tasks[i] = Explodables[i].ExplodeAsync();
148 await UniTask.WhenAll(tasks);
149
150 IsExploded = !IsExploded;

185 private ExplodablePositions SetPartPositionPoints(ExplodablePart explodable, Transform parent = null)
186 {
187 //Create a Transform to the original & exploded position and add them to the dictionary for further updates.
188 Transform originalPoint;
189 Transform explodedPoint;
190 if (pointsPool.Count > 1)
191 {
192 originalPoint = pointsPool.Dequeue();
193 explodedPoint = pointsPool.Dequeue();
194 originalPoint.gameObject.SetActive(true);
195 explodedPoint.gameObject.SetActive(true);
196 }
197 else if (pointsPool.Count == 1)
198 {
199 originalPoint = pointsPool.Dequeue();
200 originalPoint.gameObject.SetActive(true);
201 explodedPoint = new GameObject().transform;
202 }
203 else
204 {
205 originalPoint = new GameObject().transform;
206 explodedPoint = new GameObject().transform;
207 }
208 originalPoint.name = explodable.name + "_original_point";
209 explodedPoint.name = explodable.name + "_exploded_point";
210 //Set the parent of the points to the parent object.
211 try
212 {
213 originalPoint.SetParent(parent);
214 explodedPoint.SetParent(parent);
215 }
216 catch (ArgumentNullException)
217 {
218 Debug.LogWarning("Parent object is null. Setting the parent to the current object.");
219 originalPoint.SetParent(this.transform);
220 explodedPoint.SetParent(this.transform);
221 }
222 //Set the position of the points.
223 originalPoint.position = explodable.OriginalPosition;
224 explodedPoint.position = explodable.ExplodedPosition;
225 //Set the rotation of the points.
226 originalPoint.rotation = explodedPoint.rotation = explodable.transform.rotation;
227
228 return new ExplodablePositions(originalPoint, explodedPoint);
229 }

278 private Vector3 CalculateExplodedPosition(ExplodablePart explodable, float? scaleFactor = null, float? hierarchyFactor = null)
279 {
280 Func<IExplodable, Vector3> getProjectionPoint = explodable => UtilityFunctions.GetProjectionPoint(Center, explodable.OriginalPosition, DirectionAxis);
281 Func<Vector3, Vector3, float> getDistance = (point1, point2) => Vector3.Distance(point1, point2);
282
283 Vector3 originalPosition = explodable.OriginalPosition;
284 Vector3 projectionPoint = getProjectionPoint(explodable);
285 Vector3 explosionDirection;
286
287 if (explodable.TryGetComponent<ExplodableModifier>(out var modifier))
288 explosionDirection = UtilityFunctions.AxisToDirection(explodable.transform, modifier.axis);
289 else
290 explosionDirection = (originalPosition - projectionPoint).normalized;
291
292 float distanceFactor = DistanceFactor == DistanceFactor.None ? 1 : (DistanceFactor == DistanceFactor.DistanceFromCenter) ?
293 getDistance(originalPosition, Center.position) : getDistance(originalPosition, projectionPoint);
294 float additionalFactors = (scaleFactor != null ? scaleFactor.Value * this.transform.localScale.magnitude : 0) + (hierarchyFactor ?? 0);
295 return originalPosition + (explosionDirection * ExplosionDistance) * (distanceFactor + additionalFactors);
296 }
