102 public static Vector3 GetProjectionPoint(Transform centerTransform, Vector3 position, Axis directionAxis)
103 {
104 Vector3 originalPosition = position;
105 Vector3 centerPosition = centerTransform.position;
106
107 Func<Transform, Axis, Vector3> getTransformDirection = (center, axis) => AxisToDirection(center, axis);
108 Func<Axis, Vector3> getPointOnLine = axis => GetPointOnLine(originalPosition, centerPosition, getTransformDirection(centerTransform, axis));
109 Func<Axis, float> getDistance = axis => Vector3.Distance(originalPosition, getPointOnLine(axis));
110
111 if (directionAxis == Axis.X || directionAxis == Axis.Y || directionAxis == Axis.Z)
112 return getPointOnLine(directionAxis);
113 else
114 {
115 Axis[] axes;
116 switch (directionAxis)
117 {
118 case Axis.YX:
119 axes = new[] { Axis.Y, Axis.X };
120 break;
121 case Axis.XZ:
122 axes = new[] { Axis.X, Axis.Z };
123 break;
124 case Axis.YZ:
125 axes = new[] { Axis.Y, Axis.Z };
126 break;
127 default:
128 axes = new[] { Axis.Y, Axis.X, Axis.Z };
129 break;
130 }
131 // Find the axis that gives the maximum distance
132 Axis axis = axes.Aggregate((a, b) => getDistance(a) >= getDistance(b) ? a : b);
133
134 return getPointOnLine(axis);
135 }
